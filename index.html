/* Register a typed integer property */
@property --counter {
  syntax: '<integer>';
  inherits: false;
  initial-value: 0;
}

/* Animate it with scroll timeline */
@keyframes count {
  from { --counter: 0; }
  to   { --counter: 100; }
}

.stat {
  /* CSS counter display */
  counter-reset: num var(--counter);
  animation: count linear forwards;

  /* ✨ Link to scroll position */
  animation-timeline: view();
  animation-range: entry 0% entry 80%;
}

.stat::before {
  content: counter(num) "%";
}

/* Marquee driven by page scroll */
@keyframes marquee {
  from { transform: translateX(0); }
  to   { transform: translateX(-50%); }
}

.track {
  animation: marquee linear;
  animation-timeline: scroll(root);
  animation-range: 0% 100%;
}
